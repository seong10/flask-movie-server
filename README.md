> use mysql;

> -- mySQL 접속위한 유저 만들기

>					-- '%' => 네트워크 통해서 접속가능하게 하겠다
> 
> grant all on movie_project_db.* to 'm_p_user'@'%';
> -- grant all on
> -- 모든 권한을 주겠다
> -- 모든권한을 준다 momo_db의 *(모든것)에 to 'memo_user' 에다가
